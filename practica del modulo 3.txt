practica 1:
# Configurar el tiempo de espera del menú de GRUB a 20 segundos
# 1. Editar el archivo de configuración de GRUB
sudo nano /etc/default/grub

# 2. Buscar la línea GRUB_TIMEOUT= y cambiar el valor a 20 segundos
# Cambia la línea a: GRUB_TIMEOUT=20

# 3. Regenerar el archivo de configuración de GRUB con los cambios realizados
sudo grub2-mkconfig -o /boot/grub2/grub.cfg

# 4. Verificar que el tiempo de espera se ha configurado correctamente
cat /etc/default/grub | grep GRUB_TIMEOUT

# Método de recuperación si se olvida la contraseña de root:
# 1. Reiniciar el sistema y acceder a GRUB.
#    - Cuando el menú de GRUB aparezca, selecciona la opción del kernel que desees, pero **NO presiones Enter**.
#    - Presiona la tecla **'e'** para editar las opciones de arranque.

# 2. En el archivo de configuración que aparece, busca la línea que comienza con 'linux16' o 'linux'.
#    Al final de esa línea, agrega lo siguiente:
#    init=/bin/bash

# 3. Presiona **Ctrl + X** o **F10** para arrancar el sistema con esa configuración.

# 4. Montar el sistema de archivos en modo lectura-escritura
mount -o remount,rw /

# 5. Cambiar la contraseña de root
passwd root

# 6. Reiniciar el sistema
reboot
